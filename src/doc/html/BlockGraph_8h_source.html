22  template<class T = unsigned int>
23  std::vector< std::pair<T,T> > cartesian_product(std::vector<T> const & v1, std::vector<T> const & v2)
24  {
25  if(v1.size() == 0 || v2.size() == 0)
26  return std::vector< std::pair<T, T> > (); //Returns empty vector
27  if(v1.size() == 1 && v2.size() == 1)
28  return {std::make_pair(v1[0], v2[0])}; //Returns vector of size 1 created at fly
29  if(v1 == v2)
30  {
31  std::vector< std::pair<T, T> > res;
32  auto it = v2.cbegin()+1;
33  std::for_each(v1.cbegin(), v1.cend()-1,
34  [&it, &v2, &res](T const & elem){
35  std::vector< std::pair<T, T> > res_partial = cartesian_product({elem}, std::vector<T> (it, v2.cend()) );
36  it++;
37  res.insert(res.end(), std::make_move_iterator(res_partial.begin()), std::make_move_iterator(res_partial.end()) );
38  }
39  );
40  return res;
41  }
42  else
43  {
44  std::vector< std::pair<T, T> > res(v1.size()*v2.size());
45  int i = 0;
46  std::for_each(v1.cbegin(), v1.cend(), [&res, &v2, &i](T const & val_a) {
47  //First lambda -> has a for_each loop in its belly
48  std::for_each(v2.cbegin(), v2.cend(), [&res, &val_a, &i](T const & val_b) {
49  //Second lambda, constructs the pair
50  res[i++] = std::make_pair(val_a, val_b);
51 
52  }); //End second lambda
53 
54  });//End first lambda
55  return res;
56  }
57  }
