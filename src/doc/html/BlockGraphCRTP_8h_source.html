138 
139  using IdxType = typename BlockGraphBaseCRTP<D,T>::IdxType;
140  using Singleton = typename BlockGraphBaseCRTP<D,T>::Singleton;
141 
142  if(pos > this->get_possible_block_links())
143  throw std::runtime_error("Requested position exceeds matrix dimension");
144 
145  Singleton sing(ptr_groups->get_pos_singleton());
146  if(pos == 0){
147  if(std::find(sing.cbegin(), sing.cend(), 0) == sing.cend())
148  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(0,0));
149  else
150  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(0,1));
151  }
152  unsigned int extra;
153  (std::find(sing.cbegin(), sing.cend(), 0) == sing.cend()) ? extra=0 : extra=1;
154  IdxType cde_old = 0;
155  IdxType cde_new = 0;
156  IdxType last( this->get_size() - 1 ) ;
157  for(IdxType h = 0; h < last; ++h){
158  cde_new = compute_diagonal_position(h+1);
159  if(pos == cde_old){
160  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(h,h));
161  }
162  if(pos == cde_new){
163  if(std::find(sing.cbegin(), sing.cend(), h+1) == sing.cend()) // (h+1) not a singleton, select diagonal
164  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(h+1,h+1));
165  else
166  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(h,last)); //last element of the row
167  }
168  if(pos > cde_old && pos < cde_new){
169  return (h == 0) ? static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(h, pos - cde_old + h + extra)) :
170  static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(h, pos - cde_old + h)) ;
171  }
172  cde_old = cde_new;
173  }
174  return static_cast<std::pair<unsigned int, unsigned int> >(std::make_pair(last,last));

817  if(idx_i == j){
818  std::vector<unsigned int> v(this->ptr_groups->get_group(j));
819  std::copy_if(v.begin(), v.end(), std::inserter(temp, temp.begin()), [i](IdxType const & idx){return !(idx==i);} );
820  }
821  else{
822  std::vector<unsigned int> v(this->ptr_groups->get_group(j)) ;
823  std::copy(v.begin(), v.end(), std::inserter(temp, temp.begin()));
824  }
825  }
826  }
827  std::vector<unsigned int> my_nbds(temp.begin(), temp.end());
828  neighbours.insert(std::make_pair(i, my_nbds));
