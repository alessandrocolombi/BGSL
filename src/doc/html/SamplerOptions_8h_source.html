183  //1) Select prior
184  std::unique_ptr< GraphPrior<GraphType, unsigned int> > prior = nullptr;
185  if(namePr == "Uniform")
186  prior = std::move( Create_GraphPrior<PriorType::Complete, PriorCategory::Uniform, GraphType, unsigned int >() );
187  else if (namePr == "Bernoulli")
188  prior = std::move( Create_GraphPrior<PriorType::Complete, PriorCategory::Bernoulli, GraphType, unsigned int >(hy.Gprior) );
189  else
190  throw std::runtime_error("Error, the only possible priors right now are: Uniform and Bernoulli");
191  //2) Select algorithm
192  if(nameGGM == "MH")
193  return Create_GGM<GGMAlgorithm::MH, GraphType, unsigned int >(prior, hy.b_K, hy.D_K, param.trGwishSampler , param.MCiterPrior, param.MCiterPost);
194  else if(nameGGM == "RJ")
195  return Create_GGM<GGMAlgorithm::RJ, GraphType, unsigned int >(prior, hy.b_K, hy.D_K, param.trGwishSampler, hy.sigmaG, param.MCiterPrior);
196  else if(nameGGM == "DRJ")
197  return Create_GGM<GGMAlgorithm::DRJ,GraphType, unsigned int >(prior, hy.b_K, hy.D_K, param.trGwishSampler, hy.sigmaG);
198  else
199  throw std::runtime_error("Error, the only possible GGM algorithm right now are: MH, RJ, DRJ");
